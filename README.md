class Person {
  constructor() {
    this.person = "stranger";
  }

  get hi() {
   return "Hi " + this.person + "!";
  }

  set hi(arg) {
     this.person = arg;
  }
}

var person = new Person();

person.hi = "d3tu";

alert(person.hi); // Hi d3tu!
